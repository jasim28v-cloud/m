from selenium import webdriver from selenium.webdriver.chrome.service import Service from selenium.webdriver.chrome.options import Options from webdriver_manager.chrome import ChromeDriverManager import time # إعداد المتصفح ليعمل في الخلفية chrome_options = Options() chrome_options.add_argument("--headless") # اجعله يفتح بدون واجهة رسومية ليكون أسرع # تشغيل المتصفح driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()), options=chrome_options) url = "https://collection.cloudinary.com/dssk5tabb/9684298eb3619a60cd7abc8b1dbbcbfe" try: print("--- جاري فتح الرابط وتحليل البيانات... ---") driver.get(url) # الانتظار قليلاً لتحميل الصفحة time.sleep(5) # التمرير لأسفل عدة مرات للتأكد من تحميل الـ 31 فيديو بالكامل for _ in range(5): driver.execute_script("window.scrollTo(0, document.body.scrollHeight);") time.sleep(2) # استخراج كل الوسوم التي تحتوي على فيديو video_elements = driver.find_elements("tag name", "video") source_elements = driver.find_elements("tag name", "source") video_links = set() # استخدام set لمنع التكرار # البحث عن الروابط المباشرة for el in video_elements + source_elements: src = el.get_attribute("src") if src and "cloudinary.com" in src: video_links.add(src) # طباعة النتائج print(f"\n✅ تم العثور على {len(video_links)} فيديو:") print("-" * 50) for i, link in enumerate(video_links, 1): print(f"فيديو {i}: {link}") print("-" * 50) finally: driver.quit()
